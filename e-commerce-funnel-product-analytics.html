funnel_sql = """
SELECT
    SUM(sessions)         AS sessions,
    SUM(reached_product)  AS reached_product,
    SUM(reached_cart)     AS reached_cart,
    SUM(reached_shipping) AS reached_shipping,
    SUM(reached_billing)  AS reached_billing,
    SUM(converted)        AS converted
FROM mart_funnel_trend
"""
 
funnel_raw = con.execute(funnel_sql).df()
 
# Reshape to long format for funnel chart
stages = ['sessions', 'reached_product', 'reached_cart',
          'reached_shipping', 'reached_billing', 'converted']
labels = ['Sessions', 'Reached Product', 'Reached Cart',
          'Reached Shipping', 'Reached Billing', 'Converted']
values = [int(funnel_raw[s].iloc[0]) for s in stages]
 
fig = go.Figure(go.Funnel(
    y=labels,
    x=values,
    textposition='inside',
    textinfo='value+percent initial',
    marker=dict(color=[
        '#1a1a1a', '#dc2626', '#6b7280',
        '#9ca3af', '#b8bec6', '#16a34a'
    ]),
    connector=dict(line=dict(color='#e8e8e2', width=1))
))
 
fig.update_layout(
    title='Customer Conversion Funnel — 7 Stage Analysis',
    font_family='monospace',
    title_font_size=14,
    template='plotly_white'
)
 
fig.add_annotation(
    text='⚠ Biggest drop: 56% lost before product page',
    x=1, y=0.79,
    xref='paper', yref='paper',
    showarrow=False,
    font=dict(color='#dc2626', size=11),
    bgcolor='#fef2f2',

cohort_sql = """
SELECT
    cohort_month,
    months_since_first,
    AVG(active_users) AS avg_active_users
FROM mart_cohort_retention
GROUP BY cohort_month, months_since_first
ORDER BY cohort_month, months_since_first
"""
 
cohort = con.execute(cohort_sql).df()
cohort['cohort_month'] = pd.to_datetime(cohort['cohort_month'])
cohort['cohort_label'] = cohort['cohort_month'].dt.strftime('%b %Y')
 
# Normalise to % of month-0 users
cohort['month_0'] = cohort.groupby('cohort_label')['avg_active_users'].transform('first')
cohort['retention_pct'] = (cohort['avg_active_users'] / cohort['month_0'] * 100).round(1)
 
fig = px.line(
    cohort,
    x='months_since_first',
    y='retention_pct',
    color='cohort_label',
    title='Cohort Retention Curves — % of Users Returning by Month',
    labels={
        'months_since_first': 'Months Since First Visit',
        'retention_pct': 'Retention (%)',
        'cohort_label': 'Cohort'
    },
    template='plotly_white'
)
 
fig.update_traces(opacity=0.5, line=dict(width=1.2))
fig.update_layout(
    font_family='monospace',
    title_font_size=14,
    showlegend=False,  # too many cohorts — hides cleanly
    hovermode='x unified'
)
 
fig.add_annotation(
    text='Retention cliff at Month 1 — ~90% of users do not return',
    x=1, y=10,
    showarrow=True,
    arrowhead=2,
    arrowcolor='#dc2626',
    font=dict(color='#dc2626', size=11),
    ax=160, ay=-40
)
 
fig.show()

product_sql = """
SELECT
    r.product_name,
    ROUND(AVG(r.refund_rate_pct), 1)     AS refund_rate_pct,
    ROUND(AVG(c.crosssell_rate_pct), 1)  AS crosssell_rate_pct
FROM      mart_refund_rates r
LEFT JOIN mart_crosssell    c
       ON r.product_name = c.entry_product
GROUP BY r.product_name
ORDER BY crosssell_rate_pct DESC
"""

product = con.execute(product_sql).df()

fig = make_subplots(
    rows=2, cols=1,
    subplot_titles=(
        'Refund Rate by Product (%)',
        'Cross-sell Rate by Product (%)'
    ),
    vertical_spacing=0.18
)

colors_refund = ['#dc2626' if r >= 5 else '#d97706' if r >= 3 else '#16a34a'
                 for r in product['refund_rate_pct']]

fig.add_trace(go.Bar(
    x=product['refund_rate_pct'],
    y=product['product_name'],
    orientation='h',
    marker_color=colors_refund,
    name='Refund Rate',
    text=product['refund_rate_pct'].astype(str) + '%',
    textposition='outside'
), row=1, col=1)

colors_cross = ['#16a34a' if c >= 80 else '#374151'
                for c in product['crosssell_rate_pct']]

fig.add_trace(go.Bar(
    x=product['crosssell_rate_pct'],
    y=product['product_name'],
    orientation='h',
    marker_color=colors_cross,
    name='Cross-sell Rate',
    text=product['crosssell_rate_pct'].astype(str) + '%',
    textposition='outside'
), row=2, col=1)

fig.update_xaxes(range=[0, 9],   row=1)
fig.update_xaxes(range=[0, 115], row=2)

fig.update_layout(
    title='Product Health — Refund Rate vs Cross-sell Rate',
    font_family='monospace',
    title_font_size=14,
    showlegend=False,
    template='plotly_white',
    height=600,
    margin=dict(l=180, r=80, t=80, b=40)
)
fig.show()
